import { useState, useEffect } from "react";
import { 
  Card, 
  CardContent, 
  CardDescription, 
  CardHeader, 
  CardTitle 
} from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Textarea } from "@/components/ui/textarea";
import { useToast } from "@/hooks/use-toast";
import { useQuery, useMutation } from "@tanstack/react-query";
import { queryClient } from "@/lib/queryClient";
import { Label } from "@/components/ui/label";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { Loader2 } from "lucide-react";
import { useLocation } from "wouter";

export default function AddServicePage() {
  const { toast } = useToast();
  const [_, navigate] = useLocation();
  const [step, setStep] = useState<'search' | 'select' | 'price' | 'duration' | 'save'>('search');
  const [searchTerm, setSearchTerm] = useState("");
  const [selectedNiche, setSelectedNiche] = useState<string>("");
  const [selectedCategory, setSelectedCategory] = useState<string>("");
  const [selectedService, setSelectedService] = useState<any>(null);
  const [price, setPrice] = useState<string>("");
  const [duration, setDuration] = useState<string>("");
  const [executionTime, setExecutionTime] = useState<string>("");
  const [breakTime, setBreakTime] = useState<string>("");
  const [description, setDescription] = useState<string>("");

  // Fetch niches
  const { data: niches, isLoading: isLoadingNiches } = useQuery({
    queryKey: ['/api/niches'],
  });

  // Fetch categories based on selected niche
  const { data: categories, isLoading: isLoadingCategories } = useQuery({
    queryKey: ['/api/categories', selectedNiche],
    enabled: !!selectedNiche,
  });

  // Fetch services based on search term, niche and category
  const { data: services = [], isLoading: isLoadingServices } = useQuery({
    queryKey: ['/api/services/search', searchTerm, selectedNiche, selectedCategory],
    queryFn: async () => {
      const params = new URLSearchParams();
      if (searchTerm) params.append('q', searchTerm);
      if (selectedNiche) params.append('nicheId', selectedNiche);
      if (selectedCategory) params.append('categoryId', selectedCategory);
      
      // We're now querying services instead of templates
      const response = await fetch(`/api/services/search?${params.toString()}`);
      if (!response.ok) {
        throw new Error('Failed to fetch services');
      }
      return response.json();
    },
  });

  // Create provider service mutation
  const createServiceMutation = useMutation({
    mutationFn: async (serviceData: any) => {
      const response = await fetch('/api/provider/services', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(serviceData),
      });
      
      if (!response.ok) {
        throw new Error('Failed to create service');
      }
      
      return response.json();
    },
    onSuccess: () => {
      toast({
        title: "Serviço criado com sucesso!",
        description: "O serviço foi adicionado ao seu catálogo.",
      });
      queryClient.invalidateQueries({ queryKey: ['/api/provider/services'] });
      navigate("/provider/services");
    },
    onError: (error) => {
      toast({
        title: "Erro ao criar serviço",
        description: "Ocorreu um erro ao adicionar o serviço. Por favor, tente novamente.",
        variant: "destructive",
      });
    }
  });

  const handleSearch = () => {
    // The search happens automatically due to useQuery dependencies
  };

  const handleSelectService = (service: any) => {
    setSelectedService(service);
    setDescription(service.description || "");
    // Use suggested duration if available, otherwise default duration or empty string
    setDuration(service.suggestedDuration?.toString() || service.defaultDuration?.toString() || "");
    setExecutionTime(service.suggestedDuration?.toString() || service.defaultDuration?.toString() || "");
    setStep('price');
  };

  const handleSaveService = () => {
    if (!selectedService || !price || !duration || !executionTime) {
      toast({
        title: "Dados incompletos",
        description: "Por favor, preencha todos os campos obrigatórios.",
        variant: "destructive",
      });
      return;
    }

    const serviceData = {
      serviceId: selectedService.id,
      price: parseFloat(price),
      duration: parseInt(duration),
      executionTime: parseInt(executionTime),
      breakTime: breakTime ? parseInt(breakTime) : null,
      description,
    };

    createServiceMutation.mutate(serviceData);
  };

  const goBack = () => {
    if (step === 'select') setStep('search');
    if (step === 'price') setStep('select');
    if (step === 'duration') setStep('price');
    if (step === 'save') setStep('duration');
  };

  const goNext = () => {
    if (step === 'price') {
      if (!price) {
        toast({
          title: "Preço não definido",
          description: "Por favor, defina o preço do serviço.",
          variant: "destructive",
        });
        return;
      }
      setStep('duration');
    } 
    else if (step === 'duration') {
      if (!duration || !executionTime) {
        toast({
          title: "Duração não definida",
          description: "Por favor, defina a duração total e o tempo de execução do serviço.",
          variant: "destructive",
        });
        return;
      }
      setStep('save');
    }
  };

  return (
    <div className="container mx-auto py-6">
      <h1 className="text-3xl font-bold mb-6">Adicionar Novo Serviço</h1>

      {step === 'search' && (
        <Card>
          <CardHeader>
            <CardTitle>Buscar Serviços</CardTitle>
            <CardDescription>
              Encontre serviços para adicionar ao seu catálogo
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                <div>
                  <Label htmlFor="niche">Nicho</Label>
                  <Select value={selectedNiche} onValueChange={setSelectedNiche}>
                    <SelectTrigger id="niche">
                      <SelectValue placeholder="Selecione um nicho" />
                    </SelectTrigger>
                    <SelectContent>
                      <SelectItem value="">Todos os nichos</SelectItem>
                      {Array.isArray(niches) && niches.map((niche: any) => (
                        <SelectItem key={niche.id} value={niche.id.toString()}>
                          {niche.name}
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                </div>

                <div>
                  <Label htmlFor="category">Categoria</Label>
                  <Select 
                    value={selectedCategory} 
                    onValueChange={setSelectedCategory}
                    disabled={!selectedNiche || isLoadingCategories}
                  >
                    <SelectTrigger id="category">
                      <SelectValue placeholder="Selecione uma categoria" />
                    </SelectTrigger>
                    <SelectContent>
                      <SelectItem value="">Todas as categorias</SelectItem>
                      {Array.isArray(categories) && categories.map((category: any) => (
                        <SelectItem key={category.id} value={category.id.toString()}>
                          {category.name}
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                </div>
              </div>

              <div>
                <Label htmlFor="search">Buscar por nome</Label>
                <div className="flex space-x-2">
                  <Input
                    id="search"
                    placeholder="Digite para buscar serviços..."
                    value={searchTerm}
                    onChange={(e) => setSearchTerm(e.target.value)}
                  />
                  <Button onClick={handleSearch}>Buscar</Button>
                </div>
              </div>

              <div className="mt-6">
                <h3 className="text-lg font-medium mb-2">Serviços disponíveis</h3>
                {isLoadingServices ? (
                  <div className="flex justify-center p-6">
                    <Loader2 className="h-6 w-6 animate-spin" />
                  </div>
                ) : services?.length > 0 ? (
                  <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4">
                    {services.map((service: any) => (
                      <Card key={service.id} className="cursor-pointer hover:bg-gray-50 transition-colors"
                        onClick={() => {
                          setSelectedService(service);
                          setStep('select');
                        }}>
                        <CardHeader className="p-4">
                          <CardTitle className="text-lg">{service.name}</CardTitle>
                          <CardDescription className="line-clamp-2">
                            {service.description || "Sem descrição"}
                          </CardDescription>
                        </CardHeader>
                        <CardContent className="p-4 pt-0">
                          <div className="text-sm">
                            <p><span className="font-medium">Duração padrão:</span> {service.defaultDuration} min</p>
                            <p><span className="font-medium">Categoria:</span> {service.categoryName}</p>
                          </div>
                        </CardContent>
                      </Card>
                    ))}
                  </div>
                ) : (
                  <p className="text-center text-gray-500 py-4">
                    Nenhum serviço encontrado. Tente outros filtros ou termos de busca.
                  </p>
                )}
              </div>
            </div>
          </CardContent>
        </Card>
      )}

      {step === 'select' && selectedService && (
        <Card>
          <CardHeader>
            <CardTitle>Selecionar Serviço</CardTitle>
            <CardDescription>
              Confirme se este é o serviço que deseja adicionar
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              <div className="bg-gray-50 p-4 rounded-lg">
                <h3 className="text-xl font-bold">{selectedService.name}</h3>
                <p className="mt-2 text-gray-700">{selectedService.description}</p>
                
                <div className="mt-4 grid grid-cols-2 gap-4">
                  <div>
                    <span className="font-medium">Duração padrão:</span>
                    <p>{selectedService.defaultDuration} minutos</p>
                  </div>
                  <div>
                    <span className="font-medium">Categoria:</span>
                    <p>{selectedService.categoryName}</p>
                  </div>
                  {selectedService.nicheName && (
                    <div>
                      <span className="font-medium">Nicho:</span>
                      <p>{selectedService.nicheName}</p>
                    </div>
                  )}
                </div>
              </div>

              <div className="flex justify-between pt-4">
                <Button variant="outline" onClick={goBack}>Voltar</Button>
                <Button onClick={() => handleSelectService(selectedService)}>Continuar</Button>
              </div>
            </div>
          </CardContent>
        </Card>
      )}

      {step === 'price' && selectedService && (
        <Card>
          <CardHeader>
            <CardTitle>Definir Preço</CardTitle>
            <CardDescription>
              Defina o preço para este serviço
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              <div>
                <Label htmlFor="service-name">Nome do Serviço</Label>
                <p className="text-lg font-medium">{selectedService.name}</p>
              </div>

              <div>
                <Label htmlFor="price" className="block mb-2">Preço (R$)</Label>
                <Input
                  id="price"
                  type="number"
                  placeholder="0.00"
                  min="0"
                  step="0.01"
                  value={price}
                  onChange={(e) => setPrice(e.target.value)}
                  className="w-full"
                />
                <div className="text-sm text-gray-500 mt-1">
                  <p>Defina o valor que será cobrado do cliente por este serviço.</p>
                  
                  {/* Mostrando informações sobre a taxa de serviço específica do prestador */}
                  <div className="mt-2 p-2 bg-muted rounded-md">
                    <p className="font-medium">Informação sobre taxas:</p>
                    <p>Taxa fixa: R$ {selectedService.fixedFee || 'personalizada conforme contrato'}</p>
                    <p className="text-xs mt-1">Esta taxa é deduzida do valor total ao processar pagamentos.</p>
                  </div>
                </div>
              </div>

              <div className="flex justify-between pt-4">
                <Button variant="outline" onClick={goBack}>Voltar</Button>
                <Button onClick={goNext}>Continuar</Button>
              </div>
            </div>
          </CardContent>
        </Card>
      )}

      {step === 'duration' && selectedService && (
        <Card>
          <CardHeader>
            <CardTitle>Definir Duração</CardTitle>
            <CardDescription>
              Configure os tempos do seu serviço
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              <div>
                <Label htmlFor="duration" className="block mb-2">Duração Total (minutos)</Label>
                <Input
                  id="duration"
                  type="number"
                  placeholder="30"
                  min="1"
                  value={duration}
                  onChange={(e) => setDuration(e.target.value)}
                  className="w-full"
                />
                <p className="text-sm text-gray-500 mt-1">
                  Tempo total que o serviço ocupa na sua agenda.
                </p>
              </div>

              <div>
                <Label htmlFor="executionTime" className="block mb-2">Tempo de Execução (minutos)</Label>
                <Input
                  id="executionTime"
                  type="number"
                  placeholder="25"
                  min="1"
                  value={executionTime}
                  onChange={(e) => setExecutionTime(e.target.value)}
                  className="w-full"
                />
                <p className="text-sm text-gray-500 mt-1">
                  Tempo efetivo que você gasta realizando o serviço.
                </p>
              </div>

              <div>
                <Label htmlFor="breakTime" className="block mb-2">Tempo de Intervalo (minutos)</Label>
                <Input
                  id="breakTime"
                  type="number"
                  placeholder="5"
                  min="0"
                  value={breakTime}
                  onChange={(e) => setBreakTime(e.target.value)}
                  className="w-full"
                />
                <p className="text-sm text-gray-500 mt-1">
                  Tempo de pausa necessário após este serviço (opcional).
                </p>
              </div>

              <div className="flex justify-between pt-4">
                <Button variant="outline" onClick={goBack}>Voltar</Button>
                <Button onClick={goNext}>Continuar</Button>
              </div>
            </div>
          </CardContent>
        </Card>
      )}

      {step === 'save' && selectedTemplate && (
        <Card>
          <CardHeader>
            <CardTitle>Revisar e Salvar</CardTitle>
            <CardDescription>
              Confira os detalhes do serviço antes de salvar
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-4">
              <div className="bg-gray-50 p-4 rounded-lg">
                <h3 className="text-xl font-bold">{selectedTemplate.name}</h3>
                
                <div className="mt-4 grid grid-cols-2 gap-4">
                  <div>
                    <span className="font-medium">Preço:</span>
                    <p>R$ {parseFloat(price).toFixed(2)}</p>
                    <div className="text-xs text-muted-foreground mt-1 p-1 bg-muted/50 rounded">
                      <span className="font-medium">Taxa aplicada:</span> R$ {selectedTemplate.fixedFee || 'conforme contrato'}
                    </div>
                  </div>
                  <div>
                    <span className="font-medium">Duração Total:</span>
                    <p>{duration} minutos</p>
                  </div>
                  <div>
                    <span className="font-medium">Tempo de Execução:</span>
                    <p>{executionTime} minutos</p>
                  </div>
                  {breakTime && (
                    <div>
                      <span className="font-medium">Tempo de Intervalo:</span>
                      <p>{breakTime} minutos</p>
                    </div>
                  )}
                </div>
              </div>

              <div>
                <Label htmlFor="description" className="block mb-2">Descrição (opcional)</Label>
                <Textarea
                  id="description"
                  placeholder="Descreva os detalhes do serviço..."
                  value={description}
                  onChange={(e) => setDescription(e.target.value)}
                  rows={4}
                  className="w-full"
                />
              </div>

              <div className="flex justify-between pt-4">
                <Button variant="outline" onClick={goBack}>Voltar</Button>
                <Button 
                  onClick={handleSaveService}
                  disabled={createServiceMutation.isPending}
                >
                  {createServiceMutation.isPending ? (
                    <>
                      <Loader2 className="mr-2 h-4 w-4 animate-spin" />
                      Salvando...
                    </>
                  ) : (
                    "Salvar Serviço"
                  )}
                </Button>
              </div>
            </div>
          </CardContent>
        </Card>
      )}
    </div>
  );
}
