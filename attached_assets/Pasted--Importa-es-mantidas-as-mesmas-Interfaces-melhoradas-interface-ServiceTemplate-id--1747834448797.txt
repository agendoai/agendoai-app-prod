// Importações (mantidas as mesmas)

// Interfaces melhoradas
interface ServiceTemplate {
  id: number;
  name: string;
  description: string | null;
  icon: string | null;
  categoryId: number;
  nicheId: number | null;
  isActive: boolean;
  duration?: number;
  price?: number;
}

interface AddServicePayload {
  providerId: string;
  serviceId: number;
  executionTime: number;
  breakTime: number;
  price: number;
  duration: number;
  isActive: boolean;
}

export default function AddServicePage() {
  const { user } = useAuth();
  const { toast } = useToast();
  const queryClient = useQueryClient();
  const [location, setLocation] = useLocation();
  
  // Estados (mantidos os mesmos, mas com tipagem melhorada)
  const [currentTemplate, setCurrentTemplate] = useState<ServiceTemplate | null>(null);

  // Queries com tratamento de erro melhorado
  const { data: availableTemplates = [], isLoading: isLoadingTemplates, error: templatesError } = useQuery<ServiceTemplate[]>({
    queryKey: ['provider-services', 'available-templates'],
    queryFn: async () => {
      try {
        const response = await apiRequest('GET', '/api/provider-services/available-templates');
        if (!response.ok) throw new Error('Failed to fetch templates');
        return await response.json();
      } catch (error) {
        console.error('Error fetching templates:', error);
        throw error;
      }
    },
    enabled: !!user?.id,
  });

  // Mutation melhorada
  const addServiceMutation = useMutation({
    mutationFn: async (template: ServiceTemplate) => {
      if (!user?.id) throw new Error('User not authenticated');

      const payload: AddServicePayload = {
        providerId: user.id,
        serviceId: template.id,
        executionTime,
        breakTime,
        price,
        duration: executionTime + breakTime,
        isActive: true
      };

      const response = await apiRequest('POST', '/api/provider-services', payload);
      if (!response.ok) {
        const error = await response.json().catch(() => ({}));
        throw new Error(error.message || 'Failed to add service');
      }
      return response.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['provider-services', 'available-templates']);
      queryClient.invalidateQueries(['provider-services', 'provider', user?.id]);
      
      toast({
        title: 'Serviço adicionado',
        description: 'Serviço adicionado ao portfólio com sucesso',
      });
      
      setIsExecutionTimeDialogOpen(false);
      setIsPriceDialogOpen(false);
    },
    onError: (error: Error) => {
      toast({
        title: 'Erro',
        description: error.message,
        variant: 'destructive',
      });
    }
  });

  // Verificação de autenticação simplificada
  if (!user) {
    return (
      <PageTransition>
        <ProviderNavbar />
        <div className="container py-8 max-w-6xl">
          <Alert variant="destructive">
            <AlertCircle className="h-4 w-4" />
            <AlertTitle>Acesso não autorizado</AlertTitle>
            <AlertDescription>
              Você precisa estar autenticado como prestador para acessar esta página.
            </AlertDescription>
          </Alert>
        </div>
      </PageTransition>
    );
  }

  // Restante do componente (mantido similar, mas com tipagem melhorada)
  // ...
}
